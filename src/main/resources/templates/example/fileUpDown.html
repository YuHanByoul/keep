File Upload / Download











* application.yml 파일에 파일업로드 상세설정을 해야한다.

# 파일 업로드 상세설정
file: 
  filegrpName: 
    user_profile:                         # 파일그룹_이름(기업로고이미지)
      uploadPath: /user_profile            # 업로드 경로
      uploadFileSize: 10                   # 업로드가능 파일사이즈(MB)
      uploadFileExtsn: [png,gif,jpg,jpeg]  # 업로드가능 확장자
      isThumbnail: true                    # 썸네일 생성여부
      thumbnailWidth: 150                  # 썸네일 너비
      thumbnailHeight: 150                 # 썸네일 높이
      downloadChecker: [login]             # 다운로드가능 조건(로그인Y)
      imageSize: [150X30,180X45]           # 허용이미지 사이즈
      uploadFileTotalSize: 100             # (다중파일)업로드가능 총 파일사이즈(MB)
      uploadFileCnt: 2                     # (다중파일)업로드가능 파일 갯수
