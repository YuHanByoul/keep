            FieldError fieldError = bindingResult1.getFieldError();
            resultMap.put("msg", fieldError.getDefaultMessage());
            return resultMap;
        }
        
        if (bindingResult2.hasErrors()) {
            FieldError fieldError = bindingResult2.getFieldError();
            resultMap.put("msg", fieldError.getDefaultMessage());
            return resultMap;
