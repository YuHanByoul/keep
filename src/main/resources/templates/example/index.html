Controller/Service/Mapper/Vo 클래스 선언 & mapper.xml


* 사용자모듈에 사용되는 클래스의 경우 관리자모듈과 클래스명을 중복허용하기 위하여, Controller/Service/Mapper의 이름을 지정하고
  Controller와 Service/Vo의 경우는 Mybatis사용을 위해 Alias도 지정한다.
  mapper xml파일에서 parameterType / resultType 속성에 Vo객체의 Alias를 사용하도록 한다.
  Vo객체에 Alias를 적용할때에는 prefix(front.)이후 첫글자는 대문자를 사용하도록 한다.

* 관리자 모듈은 해당사항 없음.

@Controller("front.bbsController")
@Alias("front.bbsController")
public class BbsController {
 
    @Resource(name = "front.bbsServiceImpl")
    private BbsServiceImpl bbsService;
}

@Service("front.bbsServiceImpl")
@Alias("front.bbsServiceImpl")
public class BbsServiceImpl extends PlumAbstractServiceImpl implements BbsService {
 
    @Resource(name = "front.bbsDao")
    BbsDao bbsDao;
}

@Mapper("front.bbsDao")
public interface BbsDao {
}

@Data
@Alias("front.PstVo")
public class PstVo extends ParentRequestVo {
 
    /** 로그인사용자정보 */
    private UserVo user;
 
    /** 게시글 아이디 **/
    private int pstid;
}

<mapper namespace="com.kbrainc.plum.front.bbs.model.BbsDao">
    
    <!-- 게시물 목록을 가져온다. -->
    <select id="selectPstList"  parameterType="front.PstVo" resultType ="front.PstVo" >
        /* BbsDao.selectPstList */
        <include refid="PaginationMapper.header"/>
        SELECT * FROM TB_PST
          <include refid="PaginationMapper.footer"/>
    </select>
</mapper>
