권한여부에 따른 화면 처리


* 1. jsGrid 리스트 상세화면 권한 처리(상세화면 접근권한이 없으면 클릭하지 못하도록 한다.)

* 2. 버튼 권한 처리(버튼 클릭후 이동할 화면에 대한 권한이 없으면 버튼을 노출하지 않도록 한다.)

* 3. 탭 권한 처리(탭에 권한이 없는경우 노출 하지 않는다. 첫번쨰 탭에 권한이 없고 두번째 탭에 권한이 있다면 두번째 탭이 최초로 보여져야한다.)

* html태그에 sec 네임스페이스를 추가한다.

* script태그에 th:inline="javascript" 속성을 지정한다.

    function selectTab(tabId) {
        if(tabId == "tabCon1Link") {
            loadTabContent("/mng/member/memberModifyForm.html?userid=" + userid);
        } else if(tabId == "tabCon2Link") {
            loadTabContent("/mng/member/hhServiceHistoryForm.html?userid=" + userid);
        } else if(tabId == "tabCon3Link") {
            loadTabContent("/mng/member/notiHistoryForm.html?userid=" + userid);
        } else if(tabId == "tabCon4Link") {
            loadTabContent("/mng/member/smsReceiveHistoryForm.html?userid=" + userid);
        } else if(tabId == "tabCon5Link") {
            loadTabContent("/mng/member/emailReceiveHistoryForm.html?userid=" + userid);
        } else if(tabId == "tabCon6Link") {
            loadTabContent("/mng/member/loginHistoryForm.html?userid=" + userid);
        }
        $("#" + tabId).parent().addClass("active");
    }
 
    function initTab() {
        var tabFlag = false;
        /*[# sec:authorize-url="/mng/member/memberModifyForm.html"]*/
        if(tabFlag == false) {
            selectTab("tabCon1Link");
            tabFlag = true;
        }
        /*[/]*/
        /*[# sec:authorize-url="/mng/member/hhServiceHistoryForm.html"]*/
        if(tabFlag == false) {
            selectTab("tabCon2Link");

        }
        /*[/]*/
        /*[# sec:authorize-url="/mng/member/smsReceiveHistoryForm.html"]*/
        if(tabFlag == false) {
            selectTab("tabCon4Link");
            tabFlag = true;
        }
        /*[/]*/
        /*[# sec:authorize-url="/mng/member/emailReceiveHistoryForm.html"]*/
        if(tabFlag == false) {
            selectTab("tabCon5Link");
            tabFlag = true;
        }
        /*[/]*/
