/*[# sec:authorize-url="/mng/sms/sendSms.do"]*/
var phoneList = ["01066278970"];
if(displayWorkProgress()) {
    $.ajax({
        url: "/mng/sms/sendSms.do"
        , type: 'POST'
        , data: {type: "sms", msg: "안녕하세요,\n환경보전협회입니다.", phoneList: phoneList.join(",")} 
        , cache: false
        , success: function(response) {
            if(response.header.resultCode == 0) {
                alert("문자 발송이 완료되었습니다.");                                
            } else {
                alert("문자 발송이 실패 하였습니다.\n잠시 후 다시 시도해주세요.");
            }
            closeWorkProgress();
        }
    });
}
